-- Function لتحديث متوسط تقييم الخدمة
CREATE OR REPLACE FUNCTION update_service_rating()
RETURNS TRIGGER AS $$
DECLARE
    v_service_id BIGINT;
    v_total INTEGER;
    v_average DECIMAL(3,2);
    v_rating_5 INTEGER;
    v_rating_4 INTEGER;
    v_rating_3 INTEGER;
    v_rating_2 INTEGER;
    v_rating_1 INTEGER;
BEGIN
    -- تحديد service_id
    IF TG_OP = 'DELETE' THEN
        v_service_id := OLD.service_id;
    ELSE
        v_service_id := NEW.service_id;
    END IF;

    -- إذا لم يكن هناك service_id، نخرج
    IF v_service_id IS NULL THEN
        RETURN NEW;
    END IF;

    -- حساب الإحصائيات
    SELECT 
        COUNT(*),
        COALESCE(AVG(rating), 0),
        COUNT(*) FILTER (WHERE rating = 5),
        COUNT(*) FILTER (WHERE rating = 4),
        COUNT(*) FILTER (WHERE rating = 3),
        COUNT(*) FILTER (WHERE rating = 2),
        COUNT(*) FILTER (WHERE rating = 1)
    INTO 
        v_total,
        v_average,
        v_rating_5,
        v_rating_4,
        v_rating_3,
        v_rating_2,
        v_rating_1
    FROM reviews
    WHERE service_id = v_service_id AND is_approved = true;

    -- تحديث أو إدراج في service_ratings
    INSERT INTO service_ratings (
        service_id,
        total_reviews,
        average_rating,
        rating_5_star,
        rating_4_star,
        rating_3_star,
        rating_2_star,
        rating_1_star,
        updated_at
    ) VALUES (
        v_service_id,
        v_total,
        v_average,
        v_rating_5,
        v_rating_4,
        v_rating_3,
        v_rating_2,
        v_rating_1,
        NOW()
    )
    ON CONFLICT (service_id)
    DO UPDATE SET
        total_reviews = v_total,
        average_rating = v_average,
        rating_5_star = v_rating_5,
        rating_4_star = v_rating_4,
        rating_3_star = v_rating_3,
        rating_2_star = v_rating_2,
        rating_1_star = v_rating_1,
        updated_at = NOW();

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger لتحديث تقييم الخدمة
CREATE TRIGGER update_service_rating_trigger
AFTER INSERT OR UPDATE OR DELETE ON reviews
FOR EACH ROW
EXECUTE FUNCTION update_service_rating();

RETURNS VOID AS $$

    SET helpful_count = helpful_count + 1
    WHERE id = review_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Function للتحقق من وجود خدمة في المفضلة
CREATE OR REPLACE FUNCTION is_service_favorite(p_user_id INTEGER, p_service_id BIGINT)
RETURNS BOOLEAN AS $$
BEGIN
    RETURN EXISTS(
        SELECT 1 FROM favorites 
        WHERE user_id = p_user_id AND service_id = p_service_id
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Function للتحقق من صلاحية الكوبون
CREATE OR REPLACE FUNCTION validate_coupon(
    p_code VARCHAR(50),
    p_user_id INTEGER,
    p_amount DECIMAL(10, 2),
    p_is_vip BOOLEAN DEFAULT FALSE
)
RETURNS JSON AS $$
DECLARE
    v_coupon RECORD;
    v_usage_count INTEGER;
    v_user_usage_count INTEGER;
    v_discount DECIMAL(10, 2);
    v_result JSON;
BEGIN
    -- جلب الكوبون
    SELECT * INTO v_coupon
    FROM coupons
    WHERE code = p_code
    AND is_active = true
    AND NOW() BETWEEN start_date AND end_date;

    -- التحقق من وجود الكوبون
    IF v_coupon.id IS NULL THEN
        RETURN json_build_object(
            'valid', false,
            'message', 'كود الكوبون غير صحيح أو منتهي الصلاحية'
        );
    END IF;

    -- التحقق من VIP
    IF v_coupon.is_vip_only = true AND p_is_vip = false THEN
        RETURN json_build_object(
            'valid', false,
            'message', 'هذا الكوبون متاح فقط لعملاء VIP'
        );
    END IF;

    -- التحقق من الحد الأدنى للمبلغ
    IF p_amount < v_coupon.min_amount THEN
        RETURN json_build_object(
            'valid', false,
            'message', 'الحد الأدنى للاستفادة من هذا الكوبون هو ' || v_coupon.min_amount || ' ريال'
        );
    END IF;

    -- التحقق من عدد مرات الاستخدام الكلي
    IF v_coupon.usage_limit IS NOT NULL THEN
        SELECT COUNT(*) INTO v_usage_count
        FROM coupon_usages
        WHERE coupon_id = v_coupon.id;

        IF v_usage_count >= v_coupon.usage_limit THEN
            RETURN json_build_object(
                'valid', false,
                'message', 'تم استخدام هذا الكوبون الحد الأقصى من المرات'
            );
        END IF;
    END IF;

    -- التحقق من عدد مرات الاستخدام للمستخدم
    SELECT COUNT(*) INTO v_user_usage_count
    FROM coupon_usages
    WHERE coupon_id = v_coupon.id AND user_id = p_user_id;

    IF v_user_usage_count >= v_coupon.usage_per_user THEN
        RETURN json_build_object(
            'valid', false,
            'message', 'لقد استخدمت هذا الكوبون من قبل'
        );
    END IF;

    -- حساب الخصم
    IF v_coupon.discount_type = 'percentage' THEN
        v_discount := p_amount * (v_coupon.discount_value / 100);
        
        -- تطبيق الحد الأقصى للخصم إن وجد
        IF v_coupon.max_discount IS NOT NULL AND v_discount > v_coupon.max_discount THEN
            v_discount := v_coupon.max_discount;
        END IF;
    ELSE
        v_discount := v_coupon.discount_value;
    END IF;

    -- التأكد من أن الخصم لا يتجاوز المبلغ
    IF v_discount > p_amount THEN
        v_discount := p_amount;
    END IF;

    -- إرجاع النتيجة
    RETURN json_build_object(
        'valid', true,
        'coupon_id', v_coupon.id,
        'discount_type', v_coupon.discount_type,
        'discount_value', v_coupon.discount_value,
        'discount_amount', v_discount,
        'final_amount', p_amount - v_discount,
        'message', 'تم تطبيق الكوبون بنجاح'
    );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- دالة لحساب النقاط الحالية لأي مستخدم
CREATE OR REPLACE FUNCTION calculate_user_loyalty_points(user_id_param INTEGER)
RETURNS INTEGER AS $$
DECLARE
  total_points INTEGER;
BEGIN
  -- حساب مجموع النقاط من جدول loyalty_transactions
  SELECT COALESCE(SUM(
    CASE 
      WHEN transaction_type IN ('earned', 'bonus') THEN points_amount
      WHEN transaction_type IN ('redeemed', 'expired') THEN -points_amount
      ELSE 0
    END
  ), 0) INTO total_points
  FROM loyalty_transactions
  WHERE user_id = user_id_param;
  
  RETURN total_points;
END;
$$ LANGUAGE plpgsql;

-- دالة الـ Trigger
CREATE OR REPLACE FUNCTION update_user_loyalty_points()
RETURNS TRIGGER AS $$
DECLARE
  affected_user_id INTEGER;
BEGIN
  -- تحديد المستخدم المتأثر
  IF TG_OP = 'DELETE' THEN
    affected_user_id := OLD.user_id;
  ELSE
    affected_user_id := NEW.user_id;
  END IF;
  
  -- تحديث نقاط المستخدم في جدول users
  UPDATE users 
  SET loyalty_points = calculate_user_loyalty_points(affected_user_id),
      updated_at = NOW()
  WHERE id = affected_user_id;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 🔧 دالة لإنشاء كوبون تلقائي عند تحقيق معلم
-- ═══════════════════════════════════════════════════════════════

CREATE OR REPLACE FUNCTION create_loyalty_coupon(
  p_user_id INTEGER,
  p_milestone_id BIGINT
)
RETURNS BIGINT AS $$
DECLARE
  v_milestone RECORD;
  v_coupon_code TEXT;
  v_random_suffix TEXT;
  v_coupon_id BIGINT;
  v_start_date TIMESTAMP WITH TIME ZONE;
  v_end_date TIMESTAMP WITH TIME ZONE;
BEGIN
  -- جلب معلومات المعلم
  SELECT * INTO v_milestone
  FROM loyalty_milestones
  WHERE id = p_milestone_id AND is_active = true;
  
  IF v_milestone.id IS NULL THEN
    RETURN NULL;
  END IF;
  
  -- إنشاء كود كوبون فريد
  v_random_suffix := UPPER(SUBSTRING(MD5(RANDOM()::TEXT || NOW()::TEXT) FROM 1 FOR 6));
  v_coupon_code := v_milestone.coupon_code_prefix || '-' || v_random_suffix;
  
  -- تحديد تواريخ الصلاحية
  v_start_date := NOW();
  v_end_date := NOW() + (v_milestone.coupon_validity_days || ' days')::INTERVAL;
  
  -- إنشاء الكوبون في جدول coupons
  INSERT INTO coupons (
    code,
    discount_type,
    discount_value,
    max_discount,
    min_amount,
    usage_limit,
    usage_per_user,
    start_date,
    end_date,
    is_active,
    is_vip_only,
    description,
    description_ar
  ) VALUES (
    v_coupon_code,
    'percentage',
    v_milestone.discount_percentage,
    v_milestone.max_discount,
    0,
    1,
    1,
    v_start_date,
    v_end_date,
    true,
    CASE WHEN v_milestone.points_required >= 1000 THEN true ELSE false END,
    v_milestone.description,
    v_milestone.description_ar
  ) RETURNING id INTO v_coupon_id;
  
  -- تسجيل تحقيق المعلم
  INSERT INTO user_milestone_achievements (user_id, milestone_id, coupon_id)
  VALUES (p_user_id, p_milestone_id, v_coupon_id)
  ON CONFLICT (user_id, milestone_id) DO NOTHING;
  
  -- إنشاء إشعار للمستخدم
  BEGIN
    INSERT INTO notifications (
      user_id,
      title,
      message,
      notification_type,
      is_read,
      created_at
    ) VALUES (
      p_user_id,
      'مبروك! حصلت على كوبون خصم 🎉',
      'لقد وصلت إلى ' || v_milestone.points_required || ' نقطة ولاء! استخدم الكوبون: ' || v_coupon_code || ' للحصول على خصم ' || v_milestone.discount_percentage || '%',
      'reward',
      false,
      NOW()
    );
  EXCEPTION WHEN OTHERS THEN
    -- تجاهل أخطاء الإشعارات
    NULL;
  END;
  
  RETURN v_coupon_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- ═══════════════════════════════════════════════════════════════
-- 🔧 دالة للتحقق من المعالم وإنشاء كوبونات
-- ═══════════════════════════════════════════════════════════════

CREATE OR REPLACE FUNCTION check_and_create_milestone_coupons()
RETURNS TRIGGER AS $$
DECLARE
  milestone RECORD;
  v_coupon_id BIGINT;
BEGIN
  -- التحقق من جميع المعالم النشطة التي لم يتم تحقيقها بعد
  FOR milestone IN 
    SELECT lm.* 
    FROM loyalty_milestones lm
    WHERE lm.is_active = true 
    AND lm.points_required <= NEW.loyalty_points
    AND NOT EXISTS (
      SELECT 1 FROM user_milestone_achievements uma
      WHERE uma.user_id = NEW.id AND uma.milestone_id = lm.id
    )
    ORDER BY lm.points_required ASC
  LOOP
    -- إنشاء كوبون لهذا المعلم
    v_coupon_id := create_loyalty_coupon(NEW.id, milestone.id);
    
    IF v_coupon_id IS NOT NULL THEN
      RAISE NOTICE 'Created coupon % for user % achieving milestone %', v_coupon_id, NEW.id, milestone.id;
    END IF;
  END LOOP;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- ✅ حذف الـ Trigger القديم إذا كان موجوداً
DROP TRIGGER IF EXISTS trigger_check_milestone_coupons ON users;

-- ✅ إنشاء الـ Trigger الجديد
CREATE TRIGGER trigger_check_milestone_coupons
  AFTER UPDATE OF loyalty_points ON users
  FOR EACH ROW
  WHEN (NEW.loyalty_points > OLD.loyalty_points)
  EXECUTE FUNCTION check_and_create_milestone_coupons();

RETURNS TABLE (

  coupon_code VARCHAR(50),
  discount_percentage DECIMAL(5, 2),
  max_discount DECIMAL(10, 2),
  expires_at TIMESTAMP WITH TIME ZONE,
  days_left INTEGER,
  milestone_points INTEGER,
  is_used BOOLEAN
) AS $$
BEGIN
  RETURN QUERY
  SELECT 
    c.id AS coupon_id,
    c.code AS coupon_code,
    c.discount_value AS discount_percentage,
    c.max_discount,
    c.end_date AS expires_at,
    EXTRACT(DAY FROM (c.end_date - NOW()))::INTEGER AS days_left,
    lm.points_required AS milestone_points,
    EXISTS(SELECT 1 FROM coupon_usages cu WHERE cu.coupon_id = c.id AND cu.user_id = p_user_id) AS is_used
  FROM user_milestone_achievements uma
  JOIN loyalty_milestones lm ON uma.milestone_id = lm.id
  JOIN coupons c ON uma.coupon_id = c.id
  WHERE uma.user_id = p_user_id
  AND c.is_active = true
  AND c.end_date > NOW()
  AND NOT EXISTS(SELECT 1 FROM coupon_usages cu WHERE cu.coupon_id = c.id AND cu.user_id = p_user_id)
  ORDER BY c.created_at DESC;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- ✅ تحديث دالة حساب النقاط (فقط النقاط المكتملة)
CREATE OR REPLACE FUNCTION calculate_user_loyalty_points(user_id_param INTEGER)
RETURNS INTEGER AS $$
DECLARE
  total_points INTEGER;
BEGIN
  -- ✅ حساب فقط النقاط ذات status = 'completed'
  SELECT COALESCE(SUM(
    CASE 
      WHEN transaction_type IN ('earned', 'bonus') THEN points_amount
      WHEN transaction_type IN ('redeemed', 'expired', 'deducted') THEN -points_amount
      ELSE 0
    END
  ), 0) INTO total_points
  FROM loyalty_transactions
  WHERE user_id = user_id_param
  AND status = 'completed';  -- ✅ فقط المكتملة
  
  RETURN total_points;
END;
$$ LANGUAGE plpgsql;

-- ✅ تحديث الـ Trigger
CREATE OR REPLACE FUNCTION update_user_loyalty_points()
RETURNS TRIGGER AS $$
DECLARE
  affected_user_id INTEGER;
BEGIN
  IF TG_OP = 'DELETE' THEN
    affected_user_id := OLD.user_id;
  ELSE
    affected_user_id := NEW.user_id;
  END IF;
  
  -- تحديث نقاط المستخدم
  UPDATE users 
  SET loyalty_points = calculate_user_loyalty_points(affected_user_id),
      updated_at = NOW()
  WHERE id = affected_user_id;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- إعادة إنشاء الـ Trigger
DROP TRIGGER IF EXISTS trigger_update_loyalty_points ON loyalty_transactions;
CREATE TRIGGER trigger_update_loyalty_points
AFTER INSERT OR UPDATE OR DELETE ON loyalty_transactions
FOR EACH ROW
EXECUTE FUNCTION update_user_loyalty_points();





-- ═══════════════════════════════════════════════════════════════
-- 🎯 Trigger تلقائي لتفعيل النقاط عند إتمام الحجز
-- ═══════════════════════════════════════════════════════════════

CREATE OR REPLACE FUNCTION activate_loyalty_points_on_completion()
RETURNS TRIGGER AS $$
BEGIN
  -- عندما تتغير الحالة إلى completed
  IF NEW.status = 'completed' AND OLD.status != 'completed' THEN
    -- تفعيل النقاط المعلقة لهذا الحجز
    UPDATE loyalty_transactions
    SET status = 'completed'
    WHERE reference_type = 'appointment'
      AND reference_id = NEW.id
      AND status = 'pending';
    
    RAISE NOTICE 'Activated loyalty points for appointment #%', NEW.id;
  END IF;
  
  -- عندما يتم الإلغاء - إلغاء النقاط المعلقة
  IF NEW.status = 'cancelled' AND OLD.status != 'cancelled' THEN
    UPDATE loyalty_transactions
    SET status = 'cancelled'
    WHERE reference_type = 'appointment'
      AND reference_id = NEW.id
      AND status = 'pending';
    
    RAISE NOTICE 'Cancelled pending loyalty points for appointment #%', NEW.id;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- إنشاء الـ Trigger
DROP TRIGGER IF EXISTS trigger_activate_loyalty_points ON appointments;
CREATE TRIGGER trigger_activate_loyalty_points
  AFTER UPDATE OF status ON appointments
  FOR EACH ROW
  WHEN (NEW.status != OLD.status)
  EXECUTE FUNCTION activate_loyalty_points_on_completion();

RETURNS TRIGGER AS $$
DECLARE
  v_points_earned INTEGER;
  v_user_id INTEGER;
  v_service_name TEXT;
BEGIN
  -- عندما تتغير الحالة إلى completed
  IF NEW.status = 'completed' AND OLD.status != 'completed' THEN
    -- الحصول على معلومات النقاط
    SELECT points_amount, user_id 
    INTO v_points_earned, v_user_id
    FROM loyalty_transactions
    WHERE reference_type = 'appointment'
      AND reference_id = NEW.id
      AND status = 'pending'
    LIMIT 1;
    
    -- تفعيل النقاط المعلقة
    UPDATE loyalty_transactions
    SET status = 'completed'
    WHERE reference_type = 'appointment'
      AND reference_id = NEW.id
      AND status = 'pending';
    
    -- إرسال إشعار إذا كانت هناك نقاط
    IF v_points_earned IS NOT NULL AND v_points_earned > 0 THEN
      -- جلب اسم الخدمة
      SELECT service_name_ar 
      INTO v_service_name
      FROM appointment_services
      JOIN services ON appointment_services.service_id = services.id
      WHERE appointment_services.appointment_id = NEW.id
      LIMIT 1;
      
      -- إنشاء الإشعار
      INSERT INTO notifications (
        user_id,
        title,
        message,
        notification_type,
        is_read,
        created_at

        'مبروك! حصلت على نقاط ولاء 🎉',
        'لقد حصلت على ' || v_points_earned || ' نقطة ولاء من حجزك "' || COALESCE(v_service_name, 'الخدمة') || '"',
        'loyalty',
        false,
        NOW()
      );
      
      RAISE NOTICE 'Activated % loyalty points for appointment #% and sent notification', v_points_earned, NEW.id;
    END IF;
  END IF;
  
  -- عندما يتم الإلغاء - إلغاء النقاط المعلقة
  IF NEW.status = 'cancelled' AND OLD.status != 'cancelled' THEN
    UPDATE loyalty_transactions
    SET status = 'cancelled'
    WHERE reference_type = 'appointment'
      AND reference_id = NEW.id
      AND status = 'pending';
    
    RAISE NOTICE 'Cancelled pending loyalty points for appointment #%', NEW.id;
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
